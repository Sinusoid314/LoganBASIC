spritesOverlap(spriteName1, spriteName2)

Description

Returns true if the sprites spriteName1 and spriteName2 overlap, and false otherwise.

Usage

var deltaTime, prevTime var isDone = false hideConsole() enableCanvasBuffer() loadSpriteSheet("bubba-sheet", "../examples/images/bubba.png", 5, 2) loadSpriteSheet("ship-sheet", "../examples/images/ship-sheet.png", 2, 1) addSprite("bubba", "bubba-sheet", 10, 20) setSpriteFrameRange("bubba", 0, 3) setSpriteVelocityX("bubba", 60) setSpriteFrameRate("bubba", 8) addSprite("ship", "ship-sheet", 300, 50) setSpriteVelocityX("ship", -60) setSpriteFrameRate("ship", 8) prevTime = time() mainLoop() wait function mainLoop() if isDone then removeSprite("bubba") removeSprite("ship") unloadSpriteSheet("bubba-sheet") unloadSpriteSheet("ship-sheet") end end if deltaTime = (time() - prevTime) / 1000 prevTime = time() updateSprites(deltaTime) if spritesOverlap("bubba", "ship") then setSpriteFrameRange("bubba", 0, 0) isDone = true end if clearCanvas() drawSprites() drawCanvasBuffer(mainLoop) end function Open in Code Editor
