spriteOverlapsCircle(spriteName, circleX, circleY, circleRadius)

Description

Returns true if the sprite spriteName and the circle (circleX, circleY, circleRadius) overlap, and false otherwise.

Usage

var deltaTime, prevTime var circleX = 250 var circleY = 100 var circleRadius = 50 var isDone = false hideConsole() enableCanvasBuffer() loadSpriteSheet("bubba-sheet", "../examples/images/bubba.png", 5, 2) addSprite("bubba", "bubba-sheet", 10, 20) setSpriteFrameRange("bubba", 0, 3) setSpriteVelocityX("bubba", 60) setSpriteFrameRate("bubba", 8) prevTime = time() mainLoop() wait function mainLoop() if isDone then removeSprite("bubba") unloadSpriteSheet("bubba-sheet") end end if deltaTime = (time() - prevTime) / 1000 prevTime = time() updateSprites(deltaTime) if spriteOverlapsCircle("bubba", circleX, circleY, circleRadius) then setSpriteX("bubba", circleX - circleRadius - getSpriteDrawWidth("bubba")) setSpriteFrameRange("bubba", 0, 0) isDone = true end if clearCanvas() drawCircle(circleX, circleY, circleRadius) drawSprites() drawCanvasBuffer(mainLoop) end function Open in Code Editor
