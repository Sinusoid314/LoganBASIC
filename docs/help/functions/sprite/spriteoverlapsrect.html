spriteOverlapsRect(spriteName, rectX, rectY, rectWidth, rectHeight)

Description

Returns true if the sprite spriteName and the rectangle (rectX, rectY, rectWidth, rectHeight) overlap, and false otherwise.

Usage

var deltaTime, prevTime var rectX = 200 var rectY = 50 var rectWidth = 100 var rectHeight = 100 var isDone = false hideConsole() enableCanvasBuffer() loadSpriteSheet("bubba-sheet", "../examples/images/bubba.png", 5, 2) addSprite("bubba", "bubba-sheet", 10, 20) setSpriteFrameRange("bubba", 0, 3) setSpriteVelocityX("bubba", 60) setSpriteFrameRate("bubba", 8) prevTime = time() mainLoop() wait function mainLoop() if isDone then removeSprite("bubba") unloadSpriteSheet("bubba-sheet") end end if deltaTime = (time() - prevTime) / 1000 prevTime = time() updateSprites(deltaTime) if spriteOverlapsRect("bubba", rectX, rectY, rectWidth, rectHeight) then setSpriteX("bubba", rectX - getSpriteDrawWidth("bubba")) setSpriteFrameRange("bubba", 0, 0) isDone = true end if clearCanvas() drawRect(rectX, rectY, rectWidth, rectHeight) drawSprites() drawCanvasBuffer(mainLoop) end function
