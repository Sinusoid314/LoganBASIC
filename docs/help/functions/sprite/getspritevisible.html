getSpriteVisible(spriteName)

Description

Returns true if spriteName is visible, and false if it is hidden.

Usage

var isDone = false var rectX, rectY, rectWidth, rectHeight hideConsole() enableCanvasBuffer() loadSpriteSheet("bubba-sheet", "../examples/images/bubba.png", 5, 2) addSprite("bubba", "bubba-sheet", 50, 50) setSpriteFrameRange("bubba", 0, 3) setSpriteFrameRate("bubba", 8) rectX = getSpriteX("bubba") - 1 rectY = getSpriteY("bubba") - 1 rectWidth = getSpriteDrawWidth("bubba") + 2 rectHeight = getSpriteDrawHeight("bubba") + 2 setCanvasEvent("pointerup", onPointerUp) mainLoop() wait function mainLoop() if isDone then removeSprite("bubba") unloadSpriteSheet("bubba-sheet") end end if updateSprites() clearCanvas() drawRect(rectX, rectY, rectWidth, rectHeight, false) drawSprites() drawCanvasBuffer(mainLoop) end function function onPointerUp(pointerX, pointerY) if pointInSprite("bubba", pointerX, pointerY) then setSpriteVisible("bubba", not getSpriteVisible("bubba")) else isDone = true end if end function Open in Code Editor
